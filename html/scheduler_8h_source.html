1/*
2This scheduler is designed to maintain 2 schedules for use by the fuel and ignition systems.
3It functions by waiting for the overflow vectors from each of the timers in use to overflow, which triggers an interrupt
4
5//Technical
6Currently I am prescaling the 16-bit timers to 256 for injection and 64 for ignition. This means that the counter increments every 16us (injection) / 4uS (ignition) and will overflow every 1048576uS
7Max Period = (Prescale)*(1/Frequency)*(2^17)
8(See http://playground.arduino.cc/code/timer1)
9This means that the precision of the scheduler is 16uS (+/- 8uS of target) for fuel and 4uS (+/- 2uS) for ignition
10
11/Features
12This differs from most other schedulers in that its calls are non-recurring (IE You schedule an event at a certain time and once it has occurred, it will not reoccur unless you explicitely ask for it)
13Each timer can have only 1 callback associated with it at any given time. If you call the setCallback function a 2nd time, the original schedule will be overwritten and not occur
14
15Timer identification
16The Arduino timer3 is used for schedule 1
17The Arduino timer4 is used for schedule 2
18Both of these are 16-bit timers (ie count to 65536)
19See page 136 of the processors datasheet: http://www.atmel.com/Images/doc2549.pdf
20
21256 prescale gives tick every 16uS
22256 prescale gives overflow every 1048576uS (This means maximum wait time is 1.0485 seconds)
23
24*/

174static inline unsigned int setQueue(volatile Schedule *queue[], Schedule *schedule1, Schedule *schedule2, unsigned int CNT)
175{
176 //Create an array of all the upcoming targets, relative to the current count on the timer
177 unsigned int tmpQueue[4];
178
179 //Set the initial queue state. This order matches the tmpQueue order
180 if(schedule1->Status == OFF)
181 {
182 queue[0] = schedule2;
183 queue[1] = schedule2;
184 tmpQueue[0] = schedule2->startCompare - CNT;
185 tmpQueue[1] = schedule2->endCompare - CNT;
186 }
187 else
188 {
189 queue[0] = schedule1;
190 queue[1] = schedule1;
191 tmpQueue[0] = schedule1->startCompare - CNT;
192 tmpQueue[1] = schedule1->endCompare - CNT;
193 }
194
195 if(schedule2->Status == OFF)
196 {
197 queue[2] = schedule1;
198 queue[3] = schedule1;
199 tmpQueue[2] = schedule1->startCompare - CNT;
200 tmpQueue[3] = schedule1->endCompare - CNT;
201 }
202 else
203 {
204 queue[2] = schedule2;
205 queue[3] = schedule2;
206 tmpQueue[2] = schedule2->startCompare - CNT;
207 tmpQueue[3] = schedule2->endCompare - CNT;
208 }
209
210
211 //Sort the queues. Both queues are kept in sync.
212 //This implementes a sorting networking based on the Bose-Nelson sorting network
213 //See: http://pages.ripco.net/~jgamble/nw.html
214 #define SWAP(x,y) if(tmpQueue[y] < tmpQueue[x]) { unsigned int tmp = tmpQueue[x]; tmpQueue[x] = tmpQueue[y]; tmpQueue[y] = tmp; volatile Schedule *tmpS = queue[x]; queue[x] = queue[y]; queue[y] = tmpS; }
215 //SWAP(0, 1); //Likely not needed
216 //SWAP(2, 3); //Likely not needed
217 SWAP(0, 2);
218 SWAP(1, 3);
219 SWAP(1, 2);
220
221 //Return the next compare time in the queue
222 return tmpQueue[0] + CNT; //Return the
223}

225/*
226 * Moves all the Schedules in a queue forward one position.
227 * The current item (0) is discarded
228 * The final queue slot is set to nullSchedule to indicate that no action should be taken
229 */
230static inline unsigned int popQueue(volatile Schedule *queue[])
231{
232 queue[0] = queue[1];
233 queue[1] = queue[2];
234 queue[2] = queue[3];
235 queue[3] = &nullSchedule;
236
237 if( queue[0]->Status == PENDING ) { return queue[0]->startCompare; }
238 else { return queue[0]->endCompare; }
239}
